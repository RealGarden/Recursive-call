#include <iostream>
using namespace std;

void star(int n);
void space(int n);
int main() {

	int n, test;


	cin >> n;
	test = 4 * n - 3;
	for (int i = 0; i <= test/4; i++) {
		for (int j = 0; j < i; j++) {
			cout << "* ";
		}
		star(test - 4 * i);
		for (int j = 0; j < i; j++) {
			cout << " *";
		}
		cout << endl;
		if (i == test / 4) { break; }
		for (int j = 0; j <= i; j++) {
			cout << "* ";
		}
		space(4 * n - 7-4*i);
		for (int j = 0; j <= i; j++) {
			cout << " *";
		}
		cout << endl;
	}
	for (int i = test/4 -1; i >= 0; i--) {
		for (int j = 0; j <= i; j++) {
			cout << "* ";
		}
		space(4 * n - 7 - 4 * i);
		for (int j = 0; j <= i; j++) {
			cout << " *";
		}
		cout << endl;

		for (int j = 0; j < i; j++) {
			cout << "* ";
		}
		star(test - 4 * i);
		for (int j = 0; j < i; j++) {
			cout << " *";
		}
		cout << endl;


	}
	

	return 0;

}

void star(int n) {
	for (int i = 0; i < n; i++) {
		cout << "*";
	}

}

void space(int n) {
	for (int i = 0; i < n; i++) {
		cout << " ";
	}
}
