#include<iostream>

using namespace std;

int binary(unsigned long long int num);
int Binary[1000];

int main() {
	unsigned long long int num;

	cin >> num;
	int count=binary(num);
	
	for (int i = count;i >= 0;i--)
		cout << Binary[i];

	return 0;
}

int binary(unsigned long long int num) {
	static int count = 0;
	if (num <= 1) {
		Binary[count] = 1;
		return count;
	}
	Binary[count++] = num % 2;
	binary(num / 2);
	return count;
}
